1use std::process::Command;
2
3use anyhow::Context as _;
4use clap::Parser;
5
6use crate::build_ebpf::{build_ebpf, Architecture, Options as BuildOptions};
7
8#[derive(Debug, Parser)]
9pub struct Options {
10    /// Set the endianness of the BPF target
11    #[clap(default_value = "bpfel-unknown-none", long)]
12    pub bpf_target: Architecture,
13    /// Build and run the release target
14    #[clap(long)]
15    pub release: bool,
16    /// The command used to wrap your application
17    #[clap(short, long, default_value = "sudo -E")]
18    pub runner: String,
19    /// Arguments to pass to your application
20    #[clap(name = "args", last = true)]
21    pub run_args: Vec<String>,
22}
23
24/// Build the project
25fn build(opts: &Options) -> Result<(), anyhow::Error> {
26    let mut args = vec!["build"];
27    if opts.release {
28        args.push("--release")
29    }
30    let status = Command::new("cargo")
31        .args(&args)
32        .status()
33        .expect("failed to build userspace");
34    assert!(status.success());
35    Ok(())
36}
37
38/// Build and run the project
39pub fn run(opts: Options) -> Result<(), anyhow::Error> {
40    // build our ebpf program followed by our application
41    build_ebpf(
42        BuildOptions {
43            target: opts.bpf_target,
44            release: opts.release,
45        },
46        "ebpf-ipv4".to_string(),
47    )
48    .context("Error while building the eBPF Ipv4 program")?;
49    build(&opts).context("Error while building userspace application")?;
50
51    build_ebpf(
52        BuildOptions {
53            target: opts.bpf_target,
54            release: opts.release,
55        },
56        "ebpf-ipv6".to_string(),
57    )
58    .context("Error while building the eBPF Ipv6 program")?;
59    build(&opts).context("Error while building userspace application")?;
60
61    // profile we are building (release or debug)
62    let profile = if opts.release { "release" } else { "debug" };
63    let bin_path = format!("target/{profile}/rustiflow");
64
65    // Determine if 'realtime' is one of the arguments in run_args
66    let use_sudo = opts.run_args.iter().any(|arg| arg == "realtime") || opts.run_args.is_empty();
67
68    // configure args
69    let runner = if use_sudo {
70        opts.runner.trim().split_terminator(' ').collect()
71    } else {
72        Vec::new()
73    };
74    let mut args = runner;
75    args.push(bin_path.as_str());
76    args.extend(opts.run_args.iter().map(String::as_str)); // directly extend with run_args
77
78    // run the command
79    let status = Command::new(args.first().expect("No first argument"))
80        .args(args.iter().skip(1))
81        .status()
82        .expect("failed to run the command");
83
84    if !status.success() {
85        anyhow::bail!("Failed to run `{}`", args.join(" "));
86    }
87    Ok(())
88}
