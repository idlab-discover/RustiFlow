12struct ExportProfile {
13    clone_count: AtomicU64,
14    clone_time_ns: AtomicU64,
15    serialized_flow_count: AtomicU64,
16    serialized_bytes: AtomicU64,
17    dump_time_ns: AtomicU64,
18    write_time_ns: AtomicU64,
19}
20
21impl ExportProfile {
22    const fn new() -> Self {
23        Self {
24            clone_count: AtomicU64::new(0),
25            clone_time_ns: AtomicU64::new(0),
26            serialized_flow_count: AtomicU64::new(0),
27            serialized_bytes: AtomicU64::new(0),
28            dump_time_ns: AtomicU64::new(0),
29            write_time_ns: AtomicU64::new(0),
30        }
31    }
32
33    fn record_clone(&self, duration: Duration) {
34        self.clone_count.fetch_add(1, Ordering::Relaxed);
35        self.clone_time_ns
36            .fetch_add(duration.as_nanos() as u64, Ordering::Relaxed);
37    }
38
39    fn record_dump(&self, duration: Duration, bytes: usize) {
40        self.serialized_flow_count.fetch_add(1, Ordering::Relaxed);
41        self.serialized_bytes
42            .fetch_add(bytes as u64, Ordering::Relaxed);
43        self.dump_time_ns
44            .fetch_add(duration.as_nanos() as u64, Ordering::Relaxed);
45    }
46
47    fn record_write(&self, duration: Duration) {
48        self.write_time_ns
49            .fetch_add(duration.as_nanos() as u64, Ordering::Relaxed);
50    }
51
52    fn log_summary(&self, mode: &str) {
53        let clone_count = self.clone_count.load(Ordering::Relaxed);
54        let serialized_flow_count = self.serialized_flow_count.load(Ordering::Relaxed);
55        let serialized_bytes = self.serialized_bytes.load(Ordering::Relaxed);
56        let clone_time_ns = self.clone_time_ns.load(Ordering::Relaxed);
57        let dump_time_ns = self.dump_time_ns.load(Ordering::Relaxed);
58        let write_time_ns = self.write_time_ns.load(Ordering::Relaxed);
59
60        info!(
61            "Export breakdown for {}: clone_count={} clone_ms={:.3} serialized_flows={} serialized_bytes={} dump_ms={:.3} write_ms={:.3}",
62            mode,
63            clone_count,
64            clone_time_ns as f64 / 1_000_000.0,
65            serialized_flow_count,
66            serialized_bytes,
67            dump_time_ns as f64 / 1_000_000.0,
68            write_time_ns as f64 / 1_000_000.0,
69        );
70    }
71}

73fn export_profile() -> Option<&'static ExportProfile> {
74    static ENABLED: OnceLock<bool> = OnceLock::new();
75    static PROFILE: ExportProfile = ExportProfile::new();
76
77    if *ENABLED.get_or_init(|| env_flag("RUSTIFLOW_PROFILE_EXPORT_BREAKDOWN")) {
78        Some(&PROFILE)
79    } else {
80        None
81    }
82}

84pub fn record_clone(duration: Duration) {
85    if let Some(profile) = export_profile() {
86        profile.record_clone(duration);
87    }
88}

108fn env_flag(name: &str) -> bool {
109    matches!(
110        env::var(name).ok().as_deref(),
111        Some("1" | "true" | "TRUE" | "yes" | "YES" | "on" | "ON")
112    )
