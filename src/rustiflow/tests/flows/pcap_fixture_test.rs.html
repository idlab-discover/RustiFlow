25    async fn extract_fixture(name: &str) -> Vec<RustiFlow> {
26        let (tx, mut rx) = mpsc::channel::<RustiFlow>(64);
27
28        read_pcap_file::<RustiFlow>(&fixture_path(name), tx, 1, 3600, 120, None, 60)
29            .await
30            .expect("fixture pcap should parse successfully");
31
32        let mut flows = Vec::new();
33        while let Some(flow) = rx.recv().await {
34            flows.push(flow);
35        }
36
37        flows
38    }

41    async fn tiny_concap_tcp_syn_fixture_extracts_expected_flows() {
42        let flows = extract_fixture("nmap_tcp_syn_version.pcap").await;
43
44        assert_eq!(flows.len(), 17);
45
46        for flow in &flows {
47            assert_eq!(
48                count_csv_fields(&flow.dump()),
49                count_csv_fields(&RustiFlow::get_features())
50            );
51        }
52
53        let established_http_flow = flows
54            .iter()
55            .find(|flow| flow.basic_flow.flow_key == "192.168.126.228:54122-192.168.126.224:80-6")
56            .expect("expected established HTTP flow in fixture");
57        assert_eq!(established_http_flow.packet_len_stats.flow_count(), 10);
58        assert_eq!(
59            established_http_flow
60                .packet_len_stats
61                .fwd_packet_len
62                .get_count(),
63            5
64        );
65        assert_eq!(
66            established_http_flow
67                .packet_len_stats
68                .bwd_packet_len
69                .get_count(),
70            5
71        );
72        assert_eq!(
73            established_http_flow.basic_flow.flow_expire_cause,
74            FlowExpireCause::TcpTermination
75        );
76        assert!(established_http_flow.basic_flow.tcp_handshake_completed);
77        assert!(!established_http_flow.basic_flow.tcp_reset_before_handshake);
78        assert!(!established_http_flow.basic_flow.tcp_reset_after_handshake);
79        assert_eq!(established_http_flow.tcp_flags_stats.get_flags(), ".AP.SF");
80
81        let icmp_flow = flows
82            .iter()
83            .find(|flow| flow.basic_flow.flow_key == "192.168.126.228:0-192.168.126.224:0-1")
84            .expect("expected ICMP flow in fixture");
85        assert_eq!(icmp_flow.packet_len_stats.flow_count(), 4);
86        assert_eq!(icmp_flow.icmp_stats.get_type(), 8);
87        assert_eq!(icmp_flow.icmp_stats.get_code(), 0);
88        assert!(!icmp_flow.basic_flow.tcp_handshake_completed);
89        assert_eq!(
90            icmp_flow.basic_flow.flow_expire_cause,
91            FlowExpireCause::ExporterShutdown
92        );
93    }
