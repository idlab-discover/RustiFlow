13    fn temp_pcap_path(name: &str) -> PathBuf {
14        let unique = SystemTime::now()
15            .duration_since(UNIX_EPOCH)
16            .unwrap()
17            .as_nanos();
18        std::env::temp_dir().join(format!("rustiflow-{name}-{unique}.pcap"))
19    }
20
21    fn malformed_short_frame_pcap() -> Vec<u8> {
22        let mut bytes = Vec::new();
23
24        // pcap global header, little-endian, Ethernet linktype.
25        bytes.extend_from_slice(&0xa1b2c3d4_u32.to_le_bytes());
26        bytes.extend_from_slice(&2_u16.to_le_bytes());
27        bytes.extend_from_slice(&4_u16.to_le_bytes());
28        bytes.extend_from_slice(&0_i32.to_le_bytes());
29        bytes.extend_from_slice(&0_u32.to_le_bytes());
30        bytes.extend_from_slice(&65535_u32.to_le_bytes());
31        bytes.extend_from_slice(&1_u32.to_le_bytes());
32
33        let packet_data = [
34            0, 1, 2, 3, 4, 5, // dst mac
35            6, 7, 8, 9, 10, 11, // src mac
36            0x12, 0x34, // unsupported ethertype, forces fallback branch
37            0x99, // only one byte after ethertype, so old code could panic at [15]
38        ];
39
40        bytes.extend_from_slice(&1_u32.to_le_bytes());
41        bytes.extend_from_slice(&0_u32.to_le_bytes());
42        bytes.extend_from_slice(&(packet_data.len() as u32).to_le_bytes());
43        bytes.extend_from_slice(&(packet_data.len() as u32).to_le_bytes());
44        bytes.extend_from_slice(&packet_data);
45
46        bytes
47    }
48
49    #[tokio::test]
50    async fn short_unsupported_frame_does_not_panic_offline_reader() {
51        let path = temp_pcap_path("short-unsupported-frame");
52        fs::write(&path, malformed_short_frame_pcap()).expect("pcap should be written");
53
54        let (tx, mut rx) = mpsc::channel::<RustiFlow>(8);
55
56        let result =
57            read_pcap_file::<RustiFlow>(path.to_str().unwrap(), tx, 1, 3600, 120, None, 60).await;
58
59        fs::remove_file(&path).ok();
60
61        assert!(result.is_ok());
62        assert!(rx.try_recv().is_err());
63    }
