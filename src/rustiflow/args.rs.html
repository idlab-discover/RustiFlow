64#[derive(Serialize, Deserialize, Debug, Subcommand, Clone)]
65pub enum Commands {
66    /// Real-time feature extraction
67    Realtime {
68        /// The network interface to capture packets from
69        interface: String,
70        /// Whether to capture only ingress packets
71        #[clap(long, action = clap::ArgAction::SetTrue)]
72        ingress_only: bool,
73    },
74
75    /// Feature extraction from a pcap file
76    Pcap {
77        /// The relative path to the pcap file
78        path: String,
79    },
80}
81
82impl ToString for Commands {
83    fn to_string(&self) -> String {
84        match self {
85            Commands::Realtime {
86                interface,
87                ingress_only,
88            } => format!(
89                "Realtime/Interface: {}/Ingress only: {}",
90                interface, ingress_only
91            ),
92            Commands::Pcap { path } => format!("Pcap/Path: {}", path),
93        }
94    }
95}
