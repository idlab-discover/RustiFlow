65#[derive(Serialize, Deserialize, Debug, Subcommand, Clone)]
66pub enum Commands {
67    /// Real-time feature extraction
68    Realtime {
69        /// The network interface to capture packets from
70        interface: String,
71        /// Whether to capture only ingress packets
72        #[clap(long, action = clap::ArgAction::SetTrue)]
73        ingress_only: bool,
74    },
75
76    /// Feature extraction from a pcap file
77    Pcap {
78        /// The relative path to the pcap file
79        path: String,
80    },
81}
82
83impl fmt::Display for Commands {
84    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
85        match self {
86            Commands::Realtime {
87                interface,
88                ingress_only,
89            } => write!(
90                f,
91                "Realtime/Interface: {}/Ingress only: {}",
92                interface, ingress_only
93            ),
94            Commands::Pcap { path } => write!(f, "Pcap/Path: {}", path),
95        }
96    }
97}
