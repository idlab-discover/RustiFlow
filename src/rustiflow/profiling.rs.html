1use anyhow::{Context, Result};
2use log::info;
3use pprof::{ProfilerGuard, ProfilerGuardBuilder};
4use std::{env, fs::File, path::PathBuf};
5
6const DEFAULT_PROFILE_FREQUENCY_HZ: i32 = 99;
7
8pub struct ProfilingSession {
9    mode: &'static str,
10    flamegraph_path: Option<PathBuf>,
11    guard: Option<ProfilerGuard<'static>>,
12    usage_start: ResourceUsage,
13}
14
15impl ProfilingSession {
16    pub fn start_from_env(mode: &'static str) -> Result<Option<Self>> {
17        let flamegraph_path = env::var_os("RUSTIFLOW_PROFILE_FLAMEGRAPH").map(PathBuf::from);
18        let sampling_frequency_hz = env::var("RUSTIFLOW_PROFILE_FREQUENCY_HZ")
19            .ok()
20            .and_then(|value| value.parse::<i32>().ok())
21            .filter(|value| *value > 0)
22            .unwrap_or(DEFAULT_PROFILE_FREQUENCY_HZ);
23
24        let enabled = flamegraph_path.is_some() || env_flag("RUSTIFLOW_PROFILE_RESOURCE_SUMMARY");
25        if !enabled {
26            return Ok(None);
27        }
28
29        let guard = if flamegraph_path.is_some() {
30            Some(
31                ProfilerGuardBuilder::default()
32                    .frequency(sampling_frequency_hz)
33                    .blocklist(&["libc", "libgcc", "pthread", "vdso"])
34                    .build()
35                    .context("failed to start userspace profiler")?,
36            )
37        } else {
38            None
39        };
40
41        info!(
42            "Profiling enabled for {}: flamegraph={}, resource_summary={}, frequency_hz={}",
43            mode,
44            flamegraph_path
45                .as_ref()
46                .map(|path| path.display().to_string())
47                .unwrap_or_else(|| "disabled".to_string()),
48            env_flag("RUSTIFLOW_PROFILE_RESOURCE_SUMMARY"),
49            sampling_frequency_hz
50        );
51
52        Ok(Some(Self {
53            mode,
54            flamegraph_path,
55            guard,
56            usage_start: ResourceUsage::read()?,
57        }))
58    }
59
60    pub fn finish(self) -> Result<()> {
61        let usage_end = ResourceUsage::read()?;
62        let usage_delta = usage_end.delta_from(&self.usage_start);
63        info!(
64            "Profile summary for {}: user_cpu_ms={:.3} sys_cpu_ms={:.3} max_rss_kb={} voluntary_ctx_switches={} involuntary_ctx_switches={}",
65            self.mode,
66            usage_delta.user_cpu_us as f64 / 1_000.0,
67            usage_delta.system_cpu_us as f64 / 1_000.0,
68            usage_end.max_rss_kb,
69            usage_delta.voluntary_context_switches,
70            usage_delta.involuntary_context_switches
71        );
72
73        if let (Some(guard), Some(flamegraph_path)) = (self.guard, self.flamegraph_path) {
74            let report = guard
75                .report()
76                .build()
77                .context("failed to build flamegraph report")?;
78            let file = File::create(&flamegraph_path)
79                .with_context(|| format!("failed to create {}", flamegraph_path.display()))?;
80            report
81                .flamegraph(file)
82                .with_context(|| format!("failed to write {}", flamegraph_path.display()))?;
83            info!(
84                "Wrote userspace flamegraph for {} to {}",
85                self.mode,
86                flamegraph_path.display()
87            );
88        }
89
90        Ok(())
91    }
92}

94#[derive(Clone, Copy)]
95struct ResourceUsage {
96    user_cpu_us: i64,
97    system_cpu_us: i64,
98    max_rss_kb: i64,
99    voluntary_context_switches: i64,
100    involuntary_context_switches: i64,
101}
102
103impl ResourceUsage {
104    fn read() -> Result<Self> {
105        let mut usage = libc::rusage {
106            ru_utime: libc::timeval {
107                tv_sec: 0,
108                tv_usec: 0,
109            },
110            ru_stime: libc::timeval {
111                tv_sec: 0,
112                tv_usec: 0,
113            },
114            ru_maxrss: 0,
115            ru_ixrss: 0,
116            ru_idrss: 0,
117            ru_isrss: 0,
118            ru_minflt: 0,
119            ru_majflt: 0,
120            ru_nswap: 0,
121            ru_inblock: 0,
122            ru_oublock: 0,
123            ru_msgsnd: 0,
124            ru_msgrcv: 0,
125            ru_nsignals: 0,
126            ru_nvcsw: 0,
127            ru_nivcsw: 0,
128        };
129
130        let result = unsafe { libc::getrusage(libc::RUSAGE_SELF, &mut usage) };
131        if result != 0 {
132            return Err(std::io::Error::last_os_error()).context("getrusage failed");
133        }
134
135        Ok(Self {
136            user_cpu_us: timeval_to_us(usage.ru_utime),
137            system_cpu_us: timeval_to_us(usage.ru_stime),
138            max_rss_kb: usage.ru_maxrss,
139            voluntary_context_switches: usage.ru_nvcsw,
140            involuntary_context_switches: usage.ru_nivcsw,
141        })
142    }
143
144    fn delta_from(&self, start: &Self) -> Self {
145        Self {
146            user_cpu_us: self.user_cpu_us - start.user_cpu_us,
147            system_cpu_us: self.system_cpu_us - start.system_cpu_us,
148            max_rss_kb: self.max_rss_kb,
149            voluntary_context_switches: self.voluntary_context_switches
150                - start.voluntary_context_switches,
151            involuntary_context_switches: self.involuntary_context_switches
152                - start.involuntary_context_switches,
153        }
154    }
155}
156
157fn timeval_to_us(value: libc::timeval) -> i64 {
158    value.tv_sec * 1_000_000 + value.tv_usec
159}
160
161fn env_flag(name: &str) -> bool {
162    matches!(
163        env::var(name).ok().as_deref(),
164        Some("1" | "true" | "TRUE" | "yes" | "YES" | "on" | "ON")
165    )
