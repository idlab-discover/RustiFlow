12/// Represents a Custom Flow, encapsulating various metrics and states of a network flow.
13///
14/// As an example, this flow has one feature that represents the sum of the inter arrival times of the first 10 packets for both egress and ingress direction.
15///
16/// This struct is made so you can define your own features.
17#[derive(Clone)]
18pub struct CustomFlow {
19    /// Choose here for an existing flow type or leave the basic flow.
20    pub basic_flow: BasicFlow,
21    /// Add here the additional features.
22    pub icmp_stats: IcmpStats,
23}
24
25impl Flow for CustomFlow {
26    fn new(
27        flow_id: String,
28        ipv4_source: IpAddr,
29        port_source: u16,
30        ipv4_destination: IpAddr,
31        port_destination: u16,
32        protocol: u8,
33        timestamp_us: i64,
34    ) -> Self {
35        CustomFlow {
36            basic_flow: BasicFlow::new(
37                flow_id,
38                ipv4_source,
39                port_source,
40                ipv4_destination,
41                port_destination,
42                protocol,
43                timestamp_us,
44            ),
45            // Add here the initialization of the additional features, e.g. icmp stats.
46            icmp_stats: IcmpStats::new(),
47        }
48    }
49
50    fn update_flow(&mut self, packet: &PacketFeatures, fwd: bool) -> bool {
51        // Update the basic flow and returns true if the flow is terminated.
52        let last_timestamp_us = self.basic_flow.last_timestamp_us;
53        let is_terminated = self.basic_flow.update_flow(packet, fwd);
54
55        // Add here the update of the additional features.
56        self.icmp_stats.update(packet, fwd, last_timestamp_us);
57
58        // Return the termination status of the flow.
59        is_terminated
60    }
61
62    fn close_flow(&mut self, timestamp_us: i64, cause: FlowExpireCause) {
63        self.basic_flow.close_flow(timestamp_us, cause);
64
65        self.icmp_stats.close(timestamp_us, cause);
66    }
67
68    fn dump(&self) -> String {
69        // Add here the dump of the custom flow.
70        format!(
71            "{},{},{}",
72            self.basic_flow.flow_key,
73            self.icmp_stats.get_type(),
74            self.icmp_stats.get_code()
75        )
76    }
77
78    fn get_features() -> String {
79        // Add here the features of the custom flow.
80        format!("flow_id,icmp_type,icmp_code")
81    }
82
83    fn dump_without_contamination(&self) -> String {
84        // Add here the dump of the custom flow without contaminant features.
85        format!(
86            "{},{}",
87            self.icmp_stats.get_type(),
88            self.icmp_stats.get_code()
89        )
90    }
91
92    fn get_features_without_contamination() -> String {
93        // Add here the features of the custom flow without contaminant features.
94        format!("icmp_type,icmp_code")
95    }
96
97    fn get_first_timestamp_us(&self) -> i64 {
98        self.basic_flow.first_timestamp_us
99    }
100
101    fn is_expired(
102        &self,
103        timestamp_us: i64,
104        active_timeout: u64,
105        idle_timeout: u64,
106    ) -> (bool, FlowExpireCause) {
107        self.basic_flow
108            .is_expired(timestamp_us, active_timeout, idle_timeout)
109    }
110
111    fn flow_key(&self) -> &String {
112        &self.basic_flow.flow_key
113    }
114}
