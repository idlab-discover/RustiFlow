15/// Represents a CIDDS Flow as exported by the like-named academic network traffic dataset.
16#[derive(Clone)]
17pub struct CiddsFlow {
18    pub basic_flow: BasicFlow,
19    pub tcp_flag_stats: TcpFlagStats,
20    pub packet_stats: PacketLengthStats,
21}
22
23impl CiddsFlow {
24    fn format_protocol(&self, protocol: u8) -> &str {
25        match protocol {
26            p if p == IpNextHeaderProtocols::Tcp.0 => "TCP",
27            p if p == IpNextHeaderProtocols::Udp.0 => "UDP",
28            p if p == IpNextHeaderProtocols::Icmp.0 || p == IpNextHeaderProtocols::Icmpv6.0 => {
29                "ICMP"
30            }
31            _ => "OTHER",
32        }
33    }
34}
35
36impl Flow for CiddsFlow {
37    fn new(
38        flow_id: String,
39        ipv4_source: IpAddr,
40        port_source: u16,
41        ipv4_destination: IpAddr,
42        port_destination: u16,
43        protocol: u8,
44        timestamp_us: i64,
45    ) -> Self {
46        CiddsFlow {
47            basic_flow: BasicFlow::new(
48                flow_id,
49                ipv4_source,
50                port_source,
51                ipv4_destination,
52                port_destination,
53                protocol,
54                timestamp_us,
55            ),
56            tcp_flag_stats: TcpFlagStats::new(),
57            packet_stats: PacketLengthStats::new(),
58        }
59    }
60
61    fn update_flow(&mut self, packet: &PacketFeatures, is_fwd: bool) -> bool {
62        let last_timestamp_us = self.basic_flow.last_timestamp_us;
63        let is_terminated: bool = self.basic_flow.update_flow(packet, is_fwd);
64
65        self.tcp_flag_stats
66            .update(packet, is_fwd, last_timestamp_us);
67        self.packet_stats.update(packet, is_fwd, last_timestamp_us);
68
69        is_terminated
70    }
71
72    fn close_flow(&mut self, timestamp_us: i64, cause: FlowExpireCause) {
73        self.basic_flow.close_flow(timestamp_us, cause);
74
75        self.tcp_flag_stats.close(timestamp_us, cause);
76        self.packet_stats.close(timestamp_us, cause);
77    }
78
79    fn dump(&self) -> String {
80        format!(
81            "{},{},{},{},{},{},{},{},{},{}",
82            self.basic_flow.ip_source,
83            self.basic_flow.port_source,
84            self.basic_flow.ip_destination,
85            self.basic_flow.port_destination,
86            self.format_protocol(self.basic_flow.protocol),
87            self.basic_flow.get_first_timestamp(),
88            self.basic_flow.get_flow_duration_msec(),
89            self.packet_stats.flow_total(),
90            self.packet_stats.flow_count(),
91            self.tcp_flag_stats.get_flags(),
92        )
93    }
94
95    fn get_features() -> String {
96        [
97            "Src IP",
98            "Src Port",
99            "Dst IP",
100            "Dst Port",
101            "Proto",
102            "Date first seen",
103            "Duration",
104            "Bytes",
105            "Packets",
106            "Flags",
107        ]
108        .join(",")
109    }
110
111    fn dump_without_contamination(&self) -> String {

118            self.packet_stats.flow_total(),

121        )
122    }

124    fn get_features_without_contamination() -> String {

135    }

137    fn get_first_timestamp_us(&self) -> i64 {

141    fn is_expired(
