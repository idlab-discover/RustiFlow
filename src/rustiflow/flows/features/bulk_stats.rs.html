5const MIN_BULK_PACKETS: u32 = 4;
6const BULK_IDLE_MS: i64 = 1000;
7
8#[derive(Clone)]
9pub struct BulkState {
10    // Tracks "bulk in progress" state
11    pub start_time: i64,
12    pub last_time: i64,
13    pub packet_count: u32,
14    pub size: u32,
15}
16
17impl BulkState {
18    pub fn new(start_time: i64, packet_len: u16) -> Self {
19        Self {
20            start_time,
21            last_time: start_time,
22            packet_count: 1,
23            size: packet_len as u32,
24        }
25    }
26
27    pub fn update(&mut self, packet_len: u16, timestamp: i64) {
28        self.packet_count += 1;
29        self.size += packet_len as u32;
30        self.last_time = timestamp;
31    }
32}

34#[derive(Clone)]
35pub struct BulkStats {
36    pub fwd_bulk_payload_size: FeatureStats,
37    pub fwd_bulk_packets: FeatureStats,
38    pub fwd_bulk_duration: FeatureStats,
39    // Tracks "bulk in progress" state
40    fwd_bulk_state: Option<BulkState>,
41
42    pub bwd_bulk_payload_size: FeatureStats,
43    pub bwd_bulk_packets: FeatureStats,
44    pub bwd_bulk_duration: FeatureStats,
45    // Tracks "bulk in progress" state
46    bwd_bulk_state: Option<BulkState>,
47}
48
49impl BulkStats {
50    pub fn new() -> Self {
51        BulkStats {
52            fwd_bulk_payload_size: FeatureStats::new(),
53            fwd_bulk_packets: FeatureStats::new(),
54            fwd_bulk_duration: FeatureStats::new(),
55            fwd_bulk_state: None,
56
57            bwd_bulk_payload_size: FeatureStats::new(),
58            bwd_bulk_packets: FeatureStats::new(),
59            bwd_bulk_duration: FeatureStats::new(),
60            bwd_bulk_state: None,
61        }
62    }
63
64    /// Get the rate of bulk bytes in the forward direction per second.
65    pub fn fwd_bulk_rate(&self) -> f64 {
66        if self.fwd_bulk_duration.get_count() == 0 {
67            return 0.0;
68        }
69        self.fwd_bulk_payload_size.get_total() / (self.fwd_bulk_duration.get_total() / 1_000_000.0)
70    }
71
72    /// Get the rate of bulk bytes in the backward direction per second.
73    pub fn bwd_bulk_rate(&self) -> f64 {
74        if self.bwd_bulk_duration.get_count() == 0 {
75            return 0.0;
76        }
77        self.bwd_bulk_payload_size.get_total() / (self.bwd_bulk_duration.get_total() / 1_000_000.0)
78    }
79
80    /// Finalize a BulkState (if it meets the minimum packet threshold),
81    /// adding its stats to the appropriate FeatureStats.
82    fn finalize_bulk(&mut self, bulk: BulkState, is_fwd: bool) {
83        if bulk.packet_count >= MIN_BULK_PACKETS {
84            let duration_ms = bulk.last_time - bulk.start_time;
85
86            if is_fwd {
87                self.fwd_bulk_packets.add_value(bulk.packet_count as f64);
88                self.fwd_bulk_payload_size.add_value(bulk.size as f64);
89                self.fwd_bulk_duration.add_value(duration_ms as f64);
90            } else {
91                self.bwd_bulk_packets.add_value(bulk.packet_count as f64);
92                self.bwd_bulk_payload_size.add_value(bulk.size as f64);
93                self.bwd_bulk_duration.add_value(duration_ms as f64);
94            }
95        }
96    }
97
98    /// Finalize the current bulk (if any) for the given direction.
99    pub fn finalize_current_bulk(&mut self, is_fwd: bool) {
100        let old_bulk = if is_fwd {
101            self.fwd_bulk_state.take()
102        } else {
103            self.bwd_bulk_state.take()
104        };
105
106        if let Some(bulk) = old_bulk {
107            self.finalize_bulk(bulk, is_fwd);
108        }
109    }
110}

112impl FlowFeature for BulkStats {
113    /// Update the bulk logic for a given packet and direction.
114    /// Logic adopted from CICFlowMeter's bulk feature, but adjusted for capturing stats on bulk finish.
115    fn update(&mut self, packet: &PacketFeatures, is_forward: bool, _last_timestamp: i64) {
116        let current_ts = packet.timestamp_us / 1000;
117        // 1. Skip zero-length packets
118        let packet_len = packet.length;
119        if packet_len <= 0 {
120            return;
121        }
122
123        // Pick which bulk_state Option we modify (fwd or bwd).
124        let bulk_opt = if is_forward {
125            &mut self.fwd_bulk_state
126        } else {
127            &mut self.bwd_bulk_state
128        };
129
130        // We'll store any finished bulk here, finalize it once we drop `bulk_opt`
131        let mut old_bulk: Option<BulkState> = None;
132
133        match bulk_opt.as_mut() {
134            // 2. If there's an existing bulk in progress:
135            Some(current_bulk) => {
136                let gap_ms = current_ts - current_bulk.last_time;
137
138                if gap_ms > BULK_IDLE_MS {
139                    // The old bulk has ended -> remove it
140                    old_bulk = bulk_opt.take();
141                    // Start a fresh bulk with the current packet
142                    *bulk_opt = Some(BulkState::new(current_ts, packet.length));
143                } else {
144                    // Continue the same bulk
145                    current_bulk.update(packet.length, current_ts);
146                }
147            }
148            // 3. If there's no bulk in progress, start one
149            None => {
150                *bulk_opt = Some(BulkState::new(current_ts, packet.length));
151                // Finalize bulk in the other direction since we're starting a new bulk in this direction
152                self.finalize_current_bulk(!is_forward);
153            }
154        }
155
156        // 4. Now that we're done borrowing `bulk_opt`,
157        //    we can safely finalize the old bulk.
158        if let Some(bulk) = old_bulk {
159            self.finalize_bulk(bulk, is_forward);
160        }
161    }
162
163    fn close(&mut self, _last_timestamp: i64, _cause: FlowExpireCause) {
164        // Finalize any active bulks in both directions
165        self.finalize_current_bulk(true);
166        self.finalize_current_bulk(false);
167    }
168
169    fn dump(&self) -> String {
170        format!(
171            "{},{},{},{},{},{},{},{},{},{}",
172            self.fwd_bulk_rate(),
173            self.bwd_bulk_rate(),
174            self.fwd_bulk_packets.get_count(),
175            self.bwd_bulk_packets.get_count(),
176            self.fwd_bulk_packets.dump_values(),
177            self.bwd_bulk_packets.dump_values(),
178            self.fwd_bulk_payload_size.dump_values(),
179            self.bwd_bulk_payload_size.dump_values(),
180            self.fwd_bulk_duration.dump_values(),
181            self.bwd_bulk_duration.dump_values(),
182        )
183    }
184
185    fn headers() -> String {
186        format!(

188            "fwd_bulk_rate_s",
189            "bwd_bulk_rate_s",

192            FeatureStats::dump_headers("fwd_bulk_packets"),
193            FeatureStats::dump_headers("bwd_bulk_packets"),
194            FeatureStats::dump_headers("fwd_bulk_bytes"),
195            FeatureStats::dump_headers("bwd_bulk_bytes"),
196            FeatureStats::dump_headers("fwd_bulk_duration"),
197            FeatureStats::dump_headers("bwd_bulk_duration"),
198        )
199    }
