1use crate::{flows::util::FlowExpireCause, packet_features::PacketFeatures};
2
3use super::util::{FeatureStats, FlowFeature};
4
5const ACTIVE_IDLE_TIMEOUT: i64 = 5_000; // 5s
6
7#[derive(Clone)]
8pub struct ActiveIdleStats {
9    active_start: i64, // Microseconds since epoch
10    active_end: i64,
11    pub active_stats: FeatureStats,
12    pub idle_stats: FeatureStats,
13}
14
15impl ActiveIdleStats {
16    pub fn new(timestamp_us: i64) -> Self {
17        ActiveIdleStats {
18            active_start: timestamp_us,
19            active_end: timestamp_us,
20            active_stats: FeatureStats::new(),
21            idle_stats: FeatureStats::new(),
22        }
23    }
24}
25
26impl FlowFeature for ActiveIdleStats {
27    fn update(&mut self, packet: &PacketFeatures, _is_forward: bool, _last_timestamp_us: i64) {
28        let current_ts = packet.timestamp_us;
29        let duration_ms = (current_ts - self.active_end) / 1_000; // Convert to milliseconds
30
31        if duration_ms > ACTIVE_IDLE_TIMEOUT {
32            let active_duration = (self.active_end - self.active_start) / 1_000;
33            if active_duration > 0 {
34                self.active_stats.add_value(active_duration as f64);
35            }
36            self.idle_stats.add_value(duration_ms as f64);
37            self.active_start = current_ts;
38        }
39        self.active_end = current_ts;
40    }
41
42    fn close(&mut self, last_timestamp_us: i64, cause: FlowExpireCause) {
43        // If the active period is not empty, we add it to the active stats
44        let duration = self.active_end - self.active_start;
45        if duration > 0 {
46            self.active_stats.add_value(duration as f64 / 1_000.0);
47        }
48
49        // If flow expired because of inactivity, we add the idle period to the idle stats
50        if cause == FlowExpireCause::IdleTimeout {
51            self.idle_stats
52                .add_value((last_timestamp_us - self.active_end) as f64 / 1_000.0);
53        }
54    }
55
56    fn dump(&self) -> String {
57        format!(
58            "{},{}",
59            self.active_stats.dump_values(),
60            self.idle_stats.dump_values(),
61        )
62    }
63
64    fn headers() -> String {
65        format!(
66            "{},{}",
67            FeatureStats::dump_headers("active"),
68            FeatureStats::dump_headers("idle"),
69        )
70    }
71}
